% === Custom commands: Schulchronik conventions ===

% Margin note (everything in left column of original)
\newcommand{\margmark}[1]{\ledleftnote{#1}}

% Ditto / repetition mark (used in original as " for repeated words)
\newcommand{\ditto}{\textquotedbl}

% Interlinear correction: [darüber: text]
\newcommand{\darueber}[1]{\textsuperscript{\scriptsize[darüber: #1]}}

% Uncertainty marker: [?]
\newcommand{\unsicher}{\,[?]}

% Manuscript page marker (structural only — no visible output)
% The page numbers in the scans were added retrospectively and are not part of the original.
% Use in page files for documentation; does not produce output.
\newcommand{\seite}[1]{}

% Heading (centered, italic) — used for all headings (year, section, etc.)
% Must appear between \pend and \pstart in the calling code.
% Omit trailing "." from heading text.
\newcommand{\ueberschrift}[1]{%
  \bigskip
  \begin{center}
    \iffrakturheadings{\frakfont\Large #1}%
    \else{\Large\textit{#1}}%
    \fi
  \end{center}
}

% Legacy aliases
\newcommand{\jahresueberschrift}[1]{\ueberschrift{#1}}
\newcommand{\abschnitt}[1]{\ueberschrift{#1}}

% Title page heading with underline
\newcommand{\titelseite}[1]{%
  \begin{center}
    \iffrakturheadings{\frakfont\LARGE #1}%
    \else{\LARGE\textbf{#1}}%
    \fi
    \par\noindent\rule{8cm}{0.4pt}
  \end{center}
}

% Conditional line break: only active when preserving original breaks
% In reflow mode, produces a space to prevent words running together
\newcommand{\ob}{\iforiginallbreaks\\\else\space\fi}

% Original hyphenated break: for words split by a hyphen across lines
% In reflow mode, word rejoins (no hyphen, no space): "An\ohb siedlung" → "Ansiedlung"
% In original-break mode, produces hyphen + linebreak: "An-\\"
\newcommand{\ohb}{\iforiginallbreaks-\\\fi}

% Original page break marker (visual + optional forced break)
\newcommand{\originalpagebreak}{%
  \iforiginalpagebreaks\clearpage%
  \else\par\medskip\noindent\hfil\rule{3cm}{0.2pt}\hfil\par\medskip%
  \fi
}

% Horizontal rule (decorative separator in original)
\newcommand{\trenner}{\par\medskip\noindent\rule{\textwidth}{0.4pt}\par\medskip}

% Vertical emphasis line (rare, e.g., page 008)
\newcommand{\emphmargin}[1]{%
  \noindent\hspace{-1em}\rule[-\dp\strutbox]{1pt}{\baselineskip}\hspace{0.5em}#1%
}

% Annotation blocks (Hinweise, Analyse)
\newcommand{\hinweise}[1]{%
  \ifannotations
    \ifendnotes\endnote{#1}%
    \else\footnote{#1}%
    \fi
  \fi
}

\newcommand{\analyse}[1]{%
  \ifannotations
    \ifendnotes\endnote{#1}%
    \else\footnote{#1}%
    \fi
  \fi
}
